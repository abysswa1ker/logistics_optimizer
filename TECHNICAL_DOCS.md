```
    // –§–∞–∑–∞ 1: –û–ø—Ç–∏–º—ñ–∑–∞—Ü—ñ—è –ø–æ–∑–∏—Ü—ñ–π —Ç–µ—Ä–º—ñ–Ω–∞–ª—ñ–≤

        deactivate(terminal)
        new_cost = calculate_costs()
        if new_cost < current_cost:
            current_cost = new_cost  // –ó–∞–ª–∏—à–∞—î–º–æ –¥–µ–∞–∫—Ç–∏–≤–æ–≤–∞–Ω–∏–º
```

```python
    –û–¥–Ω–æ—Ç–æ—á–∫–æ–≤–∏–π –∫—Ä–æ—Å–æ–≤–µ—Ä

    –ê–ª–≥–æ—Ä–∏—Ç–º:
    1. –û–±—Ä–∞—Ç–∏ –≤–∏–ø–∞–¥–∫–æ–≤—É —Ç–æ—á–∫—É —Ä–æ–∑—Ä—ñ–∑—É
    2. –û–±–º—ñ–Ω—è—Ç–∏ —Ö–≤–æ—Å—Ç–∏ –º—ñ–∂ –±–∞—Ç—å–∫–∞–º–∏
    3. –ü–µ—Ä–µ–≤—ñ—Ä–∏—Ç–∏ –≤–∞–ª—ñ–¥–Ω—ñ—Å—Ç—å –¥—ñ—Ç–µ–π

    –ü—Ä–∏–∫–ª–∞–¥:
        parent1 = [1, 0, 1, 1, 0]
        parent2 = [0, 1, 0, 1, 1]
        point   =        ‚Üë (–ø–æ–∑–∏—Ü—ñ—è 2)

        child1  = [1, 0 | 0, 1, 1]  # –≥–æ–ª–æ–≤–∞ P1 + —Ö–≤—ñ—Å—Ç P2
        child2  = [0, 1 | 1, 1, 0]  # –≥–æ–ª–æ–≤–∞ P2 + —Ö–≤—ñ—Å—Ç P1

    Args:
        parent1, parent2: –ë–∞—Ç—å–∫—ñ–≤—Å—å–∫—ñ —Ö—Ä–æ–º–æ—Å–æ–º–∏

    Returns:
        (child1, child2): –î–≤–∞ –Ω–∞—â–∞–¥–∫–∏
    """
    n = len(parent1)

    # –í–∏–ø–∞–¥–∫–æ–≤–∞ —Ç–æ—á–∫–∞ —Ä–æ–∑—Ä—ñ–∑—É (–Ω–µ –Ω–∞ –∫—Ä–∞—è—Ö)
    point = random.randint(1, n - 1)

    # –°—Ç–≤–æ—Ä—é—î–º–æ –Ω–∞—â–∞–¥–∫—ñ–≤
    child1 = parent1[:point] + parent2[point:]
    child2 = parent2[:point] + parent1[point:]

    # –ì–∞—Ä–∞–Ω—Ç—É—î–º–æ –≤–∞–ª—ñ–¥–Ω—ñ—Å—Ç—å (–º—ñ–Ω—ñ–º—É–º 1 –∞–∫—Ç–∏–≤–Ω–∏–π)
    if sum(child1) == 0:
        child1[random.randint(0, n-1)] = 1
    if sum(child2) == 0:
        child2[random.randint(0, n-1)] = 1

    return child1, child2
```
